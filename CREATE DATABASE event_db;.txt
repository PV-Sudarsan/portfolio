CREATE DATABASE event_db;
USE event_db;
CREATE TABLE Event (
    eventid INT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    description VARCHAR(255),
    city VARCHAR(100)
);




CREATE TABLE Participant (
    playerid VARCHAR(20) PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    eventid INT,
    gender VARCHAR(10),
    year INT,

    CONSTRAINT fk_participant_event
    FOREIGN KEY (eventid) REFERENCES Event(eventid),

    CONSTRAINT chk_gender
    CHECK (gender IN ('Male', 'Female'))
);



CREATE TABLE Prizes (
    prizeid INT PRIMARY KEY,
    prize_money INT NOT NULL,
    eventid INT,
    rank_no INT,
    year INT,

    CONSTRAINT fk_prizes_event
    FOREIGN KEY (eventid) REFERENCES Event(eventid),

    CONSTRAINT chk_rank
    CHECK (rank_no IN (1, 2, 3))
);




CREATE TABLE Winners (
    prizeid INT,
    playerid VARCHAR(20),

    PRIMARY KEY (prizeid, playerid),

    CONSTRAINT fk_winners_prize
    FOREIGN KEY (prizeid) REFERENCES Prizes(prizeid),

    CONSTRAINT fk_winners_player
    FOREIGN KEY (playerid) REFERENCES Participant(playerid)
);



ALTER TABLE Participant
ADD CONSTRAINT chk_playerid_digit
CHECK (playerid REGEXP '[0-9]');



SELECT e.name
FROM Event e
JOIN Prizes pr ON e.eventid = pr.eventid
JOIN Winners w ON pr.prizeid = w.prizeid
JOIN Participant p ON w.playerid = p.playerid
GROUP BY e.eventid, e.name
HAVING COUNT(*) = SUM(CASE WHEN p.gender = 'Female' THEN 1 ELSE 0 END);


CREATE VIEW First_Prize_Winners_View AS
SELECT 
    p.name AS participant_name,
    e.name AS event_name
FROM Participant p
JOIN Winners w ON p.playerid = w.playerid
JOIN Prizes pr ON w.prizeid = pr.prizeid
JOIN Event e ON pr.eventid = e.eventid
WHERE pr.rank_no = 1;


SELECT * FROM First_Prize_Winners_View;
